//BEGIN ตรวจสอบเกรด

    INPUT score

    INPUT เกรด

    IF score >= 80 THAN

    เกรด = "A"

    ELSE IF score >= 70 THAN

    เกรด = "B"

    ELSE IF score >= 60 THAN

    เกรด = "C"

    ELSE IF score >= 50 THAN

    เกรด = "D"

    ELSE เกรด = "F"

    END IF

    OUTPUT เกรด

END


//BEGIN หาค่าสูงสุดจาก 2 ตัวเลข

    INPUT a

    INPUT b

    IF a > b THAN

    OUTPUT a

    ELSE OUTPUT b

    END IF

END


//BEGIN นับจาก 1 ถึง N

    INPUT N

    i = 1

    IF i <= N THAN

    OUTPUT i

    i = i +1

    ELSE

END
